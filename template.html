brush = alt.selection_interval(encodings=['x'])

chart = alt.Chart(cars).mark_point().encode(
    y='hp:Q',
    color=alt.condition(brush, 'cyl:N', alt.value('lightgray'))
).properties(
    width=250,
    height=250
).add_selection(
    brush
)

chart.encode(x='mpg:Q')| chart.encode(x='wt:Q')
